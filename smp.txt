def extract_and_format(data):
  """Extracts all values from a nested list and formats them.

  Args:
    data: The nested list containing the data.

  Returns:
    A string with extracted values in the format "key=value; key=value; ..."
  """
  extracted_values = []

  # Assuming the structure of your provided list (adjust indexing if needed)
  for item in data[1][0][2]:  
    for sub_item in item:
      key = sub_item[0]
      value = sub_item[1][0]
      extracted_values.append(f"{key}={value}")

  return "; ".join(extracted_values)
